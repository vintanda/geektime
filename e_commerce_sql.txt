-- 商品分类表
CREATE TABLE `product_cate` (
  `id` int(11) NOT NULL AUTO_INCREMENT COMMENT '类目id',
  `name` varchar(128) NOT NULL COMMENT '类目名称',
  `pid` int(11) NOT NULL COMMENT '父类id',
  `level` tinyint(4) DEFAULT NULL COMMENT '层级',
  `full_path` varchar(64) NOT NULL COMMENT '全路径',
  `create_time` datetime NOT NULL COMMENT '创建时间',
  `update_time` datetime NOT NULL COMMENT '更新时间',
  `sign` tinyint(4) NOT NULL DEFAULT '1' COMMENT '删除标记',
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='商品分类表'

-- 商品信息表
CREATE TABLE `product_info` (
  `id` varchar(32) NOT NULL AUTO_INCREMENT COMMENT '商品id',
  `name` varchar(256) NOT NULL DEFAULT '' COMMENT '商品名称',
  `price` double DEFAULT NULL COMMENT '价格',
  `saller_id` varchar(32) NOT NULL AUTO_INCREMENT COMMENT '店铺id',
  `info` varchar(256) DEFAULT NULL COMMENT '商品描述信息(文案图片地址)',
  `options` varchar(256) DEFAULT NULL COMMENT '可选项',
  `create_time` datetime DEFAULT NULL COMMENT '创建时间',
  `update_time` datetime DEFAULT NULL COMMENT '更新时间',
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='用户信息表'

-- 用户评价表
CREATE TABLE `evaluation` (
  `id` varchar(32) NOT NULL AUTO_INCREMENT COMMENT '评价id',
  `product_id` varchar(32) NOT NULL COMMENT '所评价的商品id',
  `user_id` varchar(32) NOT NULL COMMENT '发表评价的用户id',
  `summary` varchar(256) NOT NULL DEFAULT '' COMMENT '摘要',
  `describe` varchar(512) DEFAULT NULL COMMENT '描述信息',
  `pic_url` varchar(512) DEFAULT NULL COMMENT '图片地址',
  `create_time` datetime DEFAULT NULL COMMENT '创建时间',
  `update_time` datetime DEFAULT NULL COMMENT '更新时间',
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='用户信息表'

-- 用户表
CREATE TABLE `user` (
  `id` varchar(32) NOT NULL AUTO_INCREMENT COMMENT '用户id',
  `name` varchar(128) NOT NULL DEFAULT '' COMMENT '用户名',
  `address` varchar(512) DEFAULT NULL COMMENT '地址',
  `create_time` datetime DEFAULT NULL COMMENT '创建时间',
  `update_time` datetime DEFAULT NULL COMMENT '更新时间',
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='用户信息表'

-- 订单表
CREATE TABLE `deal_order` (
  `id` bigint(20) NOT NULL AUTO_INCREMENT COMMENT '订单id',
  `product_id` varchar(32) NOT NULL COMMENT '商品id',
  `saller_id` varchar(32) NOT NULL AUTO_INCREMENT COMMENT '店铺id',
  `deal_time` date NOT NULL COMMENT '交易时间',
  `order_status` tinyint(4) NOT NULL COMMENT '订单状态',
  `deal_snapshot` varchar(128) NOT NULL COMMENT '交易快照url',
  `create_time` datetime NOT NULL COMMENT '创建时间',
  `update_time` datetime NOT NULL COMMENT '更新时间',
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=6326 DEFAULT CHARSET=utf8mb4
